Algorithm HealthDataEntry
Input: UserHealthData[]
Output: ConfirmationMessages[]

Set entryCount = 0
Initialize ConfirmationMessages as empty list

For each entry in UserHealthData do
    entryCount = entryCount + 1

    If entry.BloodSugar >= 70 AND entry.BloodSugar <= 180
       AND entry.HeartRate > 0
       AND entry.BloodPressure != "0/0" then

        Save entry to Database
        Append "Entry " + entryCount + " Saved Successfully" 
        to ConfirmationMessages

    Else
        Append "Entry " + entryCount + " Invalid Input Detected"
        to ConfirmationMessages
    End If

End For

Return ConfirmationMessages
End Algorithm
